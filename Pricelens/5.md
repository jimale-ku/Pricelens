Shared Utility Functions

// Calculate savings
export const calculateSavings = (prices: number[]): number => {
  return Math.max(...prices) - Math.min(...prices);
};

// Find best price
export const findBestPrice = (prices: PriceInfo[]): PriceInfo => {
  return prices
    .filter(p => p.inStock)
    .sort((a, b) => a.price - b.price)[0];
};

// Format currency
export const formatCurrency = (amount: number): string => {
  return `$${amount.toFixed(2)}`;
};

// Calculate percentage savings
export const calculatePercentageSavings = (original: number, discounted: number): number => {
  return Math.round(((original - discounted) / original) * 100);
};

// Filter by search query
export const filterByQuery = (items: any[], query: string, fields: string[]): any[] => {
  if (!query) return items;
  const lowerQuery = query.toLowerCase();
  return items.filter(item =>
    fields.some(field => item[field]?.toLowerCase().includes(lowerQuery))
  );
};
